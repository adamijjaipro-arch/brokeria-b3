private async recordFailure(userId: string, step: string, ip?: string): Promise&lt;void&gt; {

   const key = `fail:${step}:${userId}`;

   const count = parseInt(await this.redis.get(key) ?? '0') + 1;

   await this.redis.set(key, count.toString(), this.lockTtl); // reset TTL ŗ chaque ťchec

   if (ip) await this.redis.incr(`fail:ip:${ip}`);            // compteur IP

   if (count &gt;= this.maxFailures) {

      await this.redis.set(`locked:${userId}`, 'true', this.lockTtl);

      this.logging.accountLocked(userId, ip);

      throw new HttpException('Compte bloquť temporairement', 423);

   }

}

Cross-Site Request Forgery (CSRF)

ModŤle d'auth : l'access token JWT est stockť en mťmoire cŰtť client (jamais dans localStorage ni dans
un cookie accessible en JS). Le cookie refresh_token est httpOnly + SameSite: strict -- il n'est pas
transmis sur des requÍtes cross-origin.

Les requÍtes API authentifiťes envoient le JWT dans le header Authorization: Bearer <token> -- un
attaquant ne peut pas forger ce header depuis un site tiers (la Same-Origin Policy l'en empÍche).

Rťsumť : l'architecture sans token CSRF explicite est sťcurisťe car :

  1. /login : saisie email + password  POST /auth/login
  2. Backend rťpond : { preAuthToken } (access refusť jusqu'au 2Ťme facteur)

Bridge Python -- `spawn` stdin/stdout

Choix de `spawn` vs `exec`

Le bridge utilise child_process.spawn avec stdio: ['pipe','pipe','pipe'] plutŰt qu'exec pour
deux raisons :

      1. Flux streaming : spawn expose les flux stdin/stdout en temps rťel --
les donnťes OHLCV (potentiellement volumineuses) sont ťcrites progressivement sans bufferiser toute la sortie
en mťmoire comme le ferait exec.

def compute_confidence(self, signals: list[dict]) -&gt; float:

"""

AgrŤge les signaux de diffťrents dťtecteurs en un score 0≠100.

Chaque signal porte un poids selon la fiabilitť du dťtecteur.

weights = {

     'candlestick': 0.25,

     'chart':         0.20,

     'indicator': 0.35, # RSI/MACD  poids fort

     'ichimoku': 0.15,

     'harmonic': 0.05,

total_score = 0.0

for sig in signals:

     w = weights.get(sig['type'], 0.1)

     total_score += sig['strength'] * w * 100

return min(100.0, max(0.0, total_score))

Alvio adopte une architecture de signaux long-only (achat uniquement), cohťrente avec les stratťgies de
trading au comptant sur les cryptomonnaies. Ce choix est dťlibťrť : sur les marchťs spot (sans levier ni vente ŗ
